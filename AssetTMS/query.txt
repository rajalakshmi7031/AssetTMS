CREATE TABLE ASSET_TB (
    ASSETID VARCHAR2(12) PRIMARY KEY,
    ASSETNAME VARCHAR2(60) NOT NULL,
    ASSETCODE VARCHAR2(30) NOT NULL,
    PURCHASEDATE DATE NOT NULL,
    CONDITION VARCHAR2(20) NOT NULL,
    DEPARTMENT VARCHAR2(50) NOT NULL,
    REMARKS VARCHAR2(100)
);
CREATE SEQUENCE ASSET_SEQ
START WITH 10
INCREMENT BY 1
MAXVALUE 99;
